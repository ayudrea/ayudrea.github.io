import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.statespace.sarimax import SARIMAX
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_absolute_error, mean_squared_error
# Convert Date column to datetime
data_combined['DATE OCC'] = pd.to_datetime(data_combined['DATE OCC'])
# Extract year, month, day, and hour
data_combined['Year'] = data_combined['DATE OCC'].dt.year
data_combined['Month'] = data_combined['DATE OCC'].dt.month
data_combined['Day'] = data_combined['DATE OCC'].dt.day
data_combined['Hour'] = data_combined['TIME OCC'] // 100
# Aggregate crime data by day and area
data_combined['Date'] = data_combined['DATE OCC'].dt.date
crime_counts = data_combined.groupby(['Date', 'AREA NAME']).size().reset_index(name='Crime Count')
# Pivot data to get a time series for each area
crime_counts_pivot = crime_counts.pivot(index='Date', columns='AREA NAME', values='Crime Count').fillna(0)
# Ensure the Date index has a frequency set
crime_counts_pivot.index = pd.to_datetime(crime_counts_pivot.index)
crime_counts_pivot = crime_counts_pivot.asfreq('D')
# Split data into training and testing sets
train_size = int(len(crime_counts_pivot) * 0.8)
train, test = crime_counts_pivot[:train_size], crime_counts_pivot[train_size:]
# Ensure train and test have the frequency set
train = train.asfreq('D')
test = test.asfreq('D')
# Train the model with area=central, which is a very good representation of the areas
area = 'Central'
train_area = train[area]
test_area = test[area]
# Use SARIMAX for yearly trends
model = SARIMAX(train_area, order=(1, 1, 1), seasonal_order=(1, 1, 1, 12))
sarima_model = model.fit()
# Forecast the value
forecast = sarima_model.forecast(steps=len(test_area))
# Evaluate the model
mae = mean_absolute_error(test_area, forecast)
print(f'Mean Absolute Error (MAE): {mae}')
# Plot the results
plt.figure(figsize=(14, 7))
plt.plot(train_area.index, train_area, label='Training Data')
plt.plot(test_area.index, test_area, label='Testing Data')
plt.plot(test_area.index, forecast, label='Forecast')
plt.title(f'Crime Forecasting for {area} area')
plt.xlabel('Date')
plt.ylabel('Crime Count')
plt.legend()
plt.show()
